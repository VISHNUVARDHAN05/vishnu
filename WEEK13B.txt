CREATE OR REPLACE TRIGGER TPROD
BEFORE INSERT OR UPDATE OR DELETE ON PRODUCTS1
FOR EACH ROW
ENABLE
DECLARE
	PROFIT_DIFF NUMBER(10);
BEGIN
	PROFIT_DIFF := :NEW.PROFIT - :OLD.PROFIT;
	DBMS_OUTPUT.PUT_LINE('NEW PROFIT:' || :NEW.PROFIT);
	DBMS_OUTPUT.PUT_LINE('OLD PROFIT:' || :OLD.PROFIT);
	DBMS_OUTPUT.PUT_LINE('PROFIT_DIFF:' || PROFIT_DIFF);
END;
/
	
Trigger created.

SQL> INSERT INTO PROD VALUES('p6','penDRIVE',200,100);
NEW PROFIT:100
OLD PROFIT:
PROFIT_DIFF:

1 row created.

SQL> UPDATE  PROD SET PROFIT=200 WHERE PID='p6';
NEW PROFIT:200
OLD PROFIT:100
PROFIT_DIFF:100

1 row updated.

SQL> DELETE PROD WHERE PID='p6';
NEW PROFIT:
OLD PROFIT:200
PROFIT_DIFF:

1 row deleted
